func EchoHandler(w http.ResponseWriter, r *http.Request) {
	if r.URL.Path != "/echo" {
		http.Error(w, "404 Not Found", http.StatusNotFound)
		return
	}

	if r.Method != http.MethodPost {
		http.Error(w, "Method Not Allowed", http.StatusMethodNotAllowed)
		return
	}

	body, err := io.ReadAll(r.Body)
	if err != nil {
		http.Error(w, "Internal Server Error", http.StatusInternalServerError)
		return
	}
	defer r.Body.Close()

	if len(body) == 0 {
		http.Error(w, "body cannot be empty", http.StatusBadRequest)
		return
	}

	w.Header().Set("Content-Type", "text/plain")
	w.Write(body)
}

func main() {
	http.HandleFunc("/echo", EchoHandler)

	err := http.ListenAndServe(":8080", nil)
	if err != nil {
		panic(err)
	}
}

1. Only allow POST
if r.Method != http.MethodPost {
	http.Error(w, "Method Not Allowed", http.StatusMethodNotAllowed)
	return
}

If someone does:

curl http://localhost:8080/echo

they get:

405 Method Not Allowed
2. Read the request body
body, err := io.ReadAll(r.Body)

Suppose the client sends:

curl -X POST -d "Hello Go" http://localhost:8080/echo

Then:

body

contains:

[]byte("Hello Go")
3. Close the body
defer r.Body.Close()

The exercise specifically asks you to place it immediately after you've successfully obtained the body stream.

A common pattern is:

body, err := io.ReadAll(r.Body)
if err != nil {
	return

defer r.Body.Close()
4. Check for an empty body
if len(body) == 0 {
	http.Error(w, "body cannot be empty", http.StatusBadRequest)
	return
}

Example:

curl -X POST -d "" http://localhost:8080/echo

Response:

400 Bad Request

body cannot be empty
5. Echo the content
w.Write(body)

If the body was:

Hello Go

the response is:

Hello Go

Exactly the same bytes.

len(body) == 0 vs body == nil

These are not always the same thing.

body, _ := io.ReadAll(r.Body)

For an empty request body, you'll typically get:

body = []byte{}

which means:

len(body) == 0  // true
body == nil     // false

Therefore:

if len(body) == 0

is the correct check.

What happens if you read r.Body twice?

Example:

body1, _ := io.ReadAll(r.Body)
body2, _ := io.ReadAll(r.Body)

After the first read:

body1 == []byte("Hello")

After the second read:

body2 == []byte{}

because the stream has already been consumed.

Think of r.Body like reading a file from start to finish. Once you've reached the end, there's nothing left to read.

Why set headers before Write?

Correct:

w.Header().Set("Content-Type", "text/plain")
w.Write(body)

Incorrect:

w.Write(body)
w.Header().Set("Content-Type", "text/plain")

When w.Write() runs for the first time, Go automatically sends the response headers to the client.

After that, changing headers usually has no effect because they have already been sent.

Think of the response order as:

Headers
↓
Status Code
↓
Body

Once the body starts going out, the headers are effectively locked.

package main

import (
	"fmt"
	"net/http"
)

/*
What happens if you use "/api" instead of "/api/"?

"/api" only matches the exact path "/api".
It will NOT automatically match "/api/v1/ping".

What does http.StripPrefix do?

It removes the specified prefix from the URL path before
passing the request to the next handler.

Without StripPrefix:
	apiMux would receive "/api/v1/ping"

With StripPrefix("/api", apiMux):
	apiMux receives "/v1/ping"

How does *http.ServeMux satisfy the Handler interface?

The Handler interface requires:

	ServeHTTP(ResponseWriter, *Request)

ServeMux implements ServeHTTP, therefore ServeMux satisfies
the Handler interface and can be passed directly to
http.ListenAndServe.
*/

func PingHandler(w http.ResponseWriter, r *http.Request) {
	fmt.Fprint(w, "pong")
}

func GreetHandler(w http.ResponseWriter, r *http.Request) {
	name := r.URL.Query().Get("name")

	if name == "" {
		fmt.Fprint(w, "Greetings, Stranger!")
		return
	}

	fmt.Fprintf(w, "Greetings, %s!", name)
}

func main() {
	apiMux := http.NewServeMux()

	apiMux.HandleFunc("/v1/ping", PingHandler)
	apiMux.HandleFunc("/v1/greet", GreetHandler)

	mainMux := http.NewServeMux()

	mainMux.Handle(
		"/api/",
		http.StripPrefix("/api", apiMux),
	)

	fmt.Println("Server running on :8080")

	err := http.ListenAndServe(":8080", mainMux)
	if err != nil {
		panic(err)
	}
}

			http.StatusBadRequest,
		)
		return
	}

	w.Header().Set("Content-Type", "text/plain")
	w.Write(body)
}

func StatusHandler(w http.ResponseWriter, r *http.Request) {
	codeStr := r.URL.Query().Get("code")

	if codeStr == "" {
		http.Error(
			w,
			"code parameter is required",
			http.StatusBadRequest,
		)
		return
	}

	code, err := strconv.Atoi(codeStr)
	if err != nil {
		http.Error(
			w,
			"code must be a valid integer",
			http.StatusBadRequest,
		)
		return
	}

	if code < 100 || code > 599 {
		http.Error(
			w,
			"code must be a valid HTTP status code (100-599)",
			http.StatusBadRequest,
		)
		return
	}

	w.WriteHeader(code)

	fmt.Fprintf(
		w,
		"Responding with status %d %s",
		code,
		http.StatusText(code),
	)
}

func HeadersHandler(w http.ResponseWriter, r *http.Request) {
	token := r.Header.Get("X-Custom-Token")

	if token == "" {
		http.Error(
			w,
			"X-Custom-Token header is missing",
			http.StatusBadRequest,
		)
		return
	}

	contentType := r.Header.Get("Content-Type")

	if contentType == "" {
		contentType = "Content-Type not provided"
	} else {
		contentType = "Content-Type: " + contentType
	}

	fmt.Fprintf(
		w,
		"Token received: %s\n%s",
		token,
		contentType,
	)
}

func RenderHandler(w http.ResponseWriter, r *http.Request) {
	title := r.URL.Query().Get("title")
	body := r.URL.Query().Get("body")

	if title == "" || body == "" {
		http.Error(
			w,
			"title and body are required",
			http.StatusBadRequest,
		)
		return
	}

	w.Header().Set("Content-Type", "text/html")

	err := tmpl.Execute(
		w,
		PageData{
			Title: title,
			Body:  body,
		},
	)

	if err != nil {
		http.Error(
			w,
			"template execution failed",
			http.StatusInternalServerError,
		)
		return
	}
}

func PingHandler(w http.ResponseWriter, r *http.Request) {
	fmt.Fprint(w, "pong")
}

func GreetHandler(w http.ResponseWriter, r *http.Request) {
	name := r.URL.Query().Get("name")

	if name == "" {
		fmt.Fprint(w, "Greetings, Stranger!")
		return
	}

	fmt.Fprintf(
		w,
		"Greetings, %s!",
		name,
	)
}

func main() {
	apiMux := http.NewServeMux()

	apiMux.HandleFunc("/v1/ping", PingHandler)
	apiMux.HandleFunc("/v1/greet", GreetHandler)

	mainMux := http.NewServeMux()

	mainMux.HandleFunc("/", HomeHandler)
	mainMux.HandleFunc("/method", MethodHandler)
	mainMux.HandleFunc("/echo", EchoHandler)
	mainMux.HandleFunc("/status", StatusHandler)
	mainMux.HandleFunc("/headers", HeadersHandler)
	mainMux.HandleFunc("/render", RenderHandler)

	mainMux.Handle(
		"/api/",
		http.StripPrefix("/api", apiMux),
	)

	fmt.Println("Server running on :8080")

	err := http.ListenAndServe(":8080", mainMux)
	if err != nil {
		panic(err)
	}
}
